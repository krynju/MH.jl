5 # This file contains the creation of Sampler objects and the associated generation of
6 # random values from them. More specifically, given the specification S of a set
7 # of values to pick from (e.g. 1:10, or "a string"), we define
8 #
9 # 1) Sampler(rng, S, ::Repetition) -> sampler
10 # 2) rand(rng, sampler) -> random value
11 #
12 # Note that the 1) is automated when the sampler is not intended to carry information,
13 # i.e. the default fall-backs SamplerType and SamplerTrivial are used.

48 function SamplerBigFloat{I}(prec::Int) where I<:FloatInterval{BigFloat}
49 nlimbs = (prec-1) ÷ bits_in_Limb + 1
50 limbs = Vector{Limb}(undef, nlimbs)
51 shift = nlimbs * bits_in_Limb - prec
52 new(prec, nlimbs, limbs, shift)
53 end

59 function _rand(rng::AbstractRNG, sp::SamplerBigFloat)
60 z = BigFloat()
61 limbs = sp.limbs
62 rand!(rng, limbs)
63 @inbounds begin
64 limbs[1] <<= sp.shift
65 randbool = iszero(limbs[end] & Limb_high_bit)
66 limbs[end] |= Limb_high_bit
67 end
68 z.sign = 1
69 GC.@preserve limbs unsafe_copyto!(z.d, pointer(limbs), sp.nlimbs)
70 (z, randbool)
71 end

167 ### BitInteger
168
169 # there are three implemented samplers for unit ranges, the two first of which
170 # assume that Float64 (i.e. 52 random bits) is the native type for the RNG:
171 # 1) "Fast" (SamplerRangeFast), which is most efficient when the underlying RNG produces
172 # rand(Float64) "fast enough".
173 # The tradeoff is faster creation of the sampler, but more consumption of entropy bits.
174 # 2) "Slow" (SamplerRangeInt) which tries to use as few entropy bits as possible, at the
175 # cost of a a bigger upfront price associated with the creation of the sampler.
176 # This sampler is most appropriate for slower random generators.
177 # 3) "Nearly Division Less" (NDL) which is generally the fastest algorithm for types of size
178 # up to 64 bits. This is the default for these types since Julia 1.5.
179 # The "Fast" algorithm can be faster than NDL when the length of the range is
180 # less than and close to a power of 2.
181
182 Sampler(::Type{<:AbstractRNG}, r::AbstractUnitRange{T},
183 ::Repetition) where {T<:Base.BitInteger64} = SamplerRangeNDL(r)
184
185 Sampler(::Type{<:AbstractRNG}, r::AbstractUnitRange{T},
186 ::Repetition) where {T<:Union{Int128,UInt128}} = SamplerRangeFast(r)

206 function SamplerRangeFast(r::AbstractUnitRange{T}, ::Type{U}) where {T,U}
207 isempty(r) && throw(ArgumentError("range must be non-empty"))
208 m = (last(r) - first(r)) % unsigned(T) % U # % unsigned(T) to not propagate sign bit
209 bw = (sizeof(U) << 3 - leading_zeros(m)) % UInt # bit-width
210 mask = ((1 % U) << bw) - (1 % U)
211 SamplerRangeFast{U,T}(first(r), bw, m, mask)
212 end

238 #### "Slow" / SamplerRangeInt
239
240 # remainder function according to Knuth, where rem_knuth(a, 0) = a
241 rem_knuth(a::UInt, b::UInt) = a % (b + (b == 0)) + a * (b == 0)
242 rem_knuth(a::T, b::T) where {T<:Unsigned} = b != 0 ? a % b : a
243
244 # maximum multiple of k <= sup decremented by one,
245 # that is 0xFFFF...FFFF if k = (typemax(T) - typemin(T)) + 1 and sup == typemax(T) - 1
246 # with intentional underflow
247 # see http://stackoverflow.com/questions/29182036/integer-arithmetic-add-1-to-uint-max-and-divide-by-n-without-overflow
248
249 # sup == 0 means typemax(T) + 1
250 maxmultiple(k::T, sup::T=zero(T)) where {T<:Unsigned} =
251 (div(sup - k, k + (k == 0))*k + k - one(k))::T
252
253 # similar but sup must not be equal to typemax(T)
254 unsafe_maxmultiple(k::T, sup::T) where {T<:Unsigned} =
255 div(sup, k + (k == 0))*k - one(k)

257 struct SamplerRangeInt{T<:Integer,U<:Unsigned} <: Sampler{T}
258 a::T # first element of the range
259 bw::Int # bit width
260 k::U # range length or zero for full range
261 u::U # rejection threshold
262 end

265 SamplerRangeInt(r::AbstractUnitRange{T}) where T<:BitInteger =

268 function SamplerRangeInt(r::AbstractUnitRange{T}, ::Type{U}) where {T,U}
269 isempty(r) && throw(ArgumentError("range must be non-empty"))
270 a = first(r)
271 m = (last(r) - first(r)) % unsigned(T) % U
272 k = m + one(U)
273 bw = (sizeof(U) << 3 - leading_zeros(m)) % Int
274 mult = if U === UInt32
275 maxmultiple(k)
276 elseif U === UInt64
277 bw <= 52 ? unsafe_maxmultiple(k, one(UInt64) << 52) :
278 maxmultiple(k)
279 else # U === UInt128
280 bw <= 52 ? unsafe_maxmultiple(k, one(UInt128) << 52) :
281 bw <= 104 ? unsafe_maxmultiple(k, one(UInt128) << 104) :
282 maxmultiple(k)
283 end
284
285 SamplerRangeInt{T,U}(a, bw, k, mult) # overflow ok
286 end

305 #### Nearly Division Less
306
307 # cf. https://arxiv.org/abs/1805.10941 (algorithm 5)
308
309 struct SamplerRangeNDL{U<:Unsigned,T} <: Sampler{T}
310 a::T # first element of the range
311 s::U # range length or zero for full range
312 end
313
314 function SamplerRangeNDL(r::AbstractUnitRange{T}) where {T}
315 isempty(r) && throw(ArgumentError("range must be non-empty"))
316 a = first(r)
317 U = uint_sup(T)
318 s = (last(r) - first(r)) % unsigned(T) % U + one(U) # overflow ok
319 # mod(-s, s) could be put in the Sampler object for repeated calls, but
320 # this would be an advantage only for very big s and number of calls
321 SamplerRangeNDL(a, s)
322 end
323
324 function rand(rng::AbstractRNG, sp::SamplerRangeNDL{U,T}) where {U,T}
325 s = sp.s
326 x = widen(rand(rng, U))
327 m = x * s
328 l = m % U
329 if l < s
330 t = mod(-s, s) # as s is unsigned, -s is equal to 2^L - s in the paper
331 while l < t
332 x = widen(rand(rng, U))
333 m = x * s
334 l = m % U
335 end
336 end
337 (s == 0 ? x : m >> (8*sizeof(U))) % T + sp.a
338 end

343 struct SamplerBigInt <: Sampler{BigInt}
344 a::BigInt # first
345 m::BigInt # range length - 1
346 nlimbs::Int # number of limbs in generated BigInt's (z ∈ [0, m])
347 nlimbsmax::Int # max number of limbs for z+a
348 mask::Limb # applied to the highest limb
349 end
350
351 function Sampler(::Type{<:AbstractRNG}, r::AbstractUnitRange{BigInt}, ::Repetition)
352 m = last(r) - first(r)
353 m < 0 && throw(ArgumentError("range must be non-empty"))
354 nd = ndigits(m, base=2)
355 nlimbs, highbits = divrem(nd, 8*sizeof(Limb))
356 highbits > 0 && (nlimbs += 1)
357 mask = highbits == 0 ? ~zero(Limb) : one(Limb)<<highbits - one(Limb)
358 nlimbsmax = max(nlimbs, abs(last(r).size), abs(first(r).size))
359 return SamplerBigInt(first(r), m, nlimbs, nlimbsmax, mask)
360 end
361
362 function rand(rng::AbstractRNG, sp::SamplerBigInt)
363 x = MPZ.realloc2(sp.nlimbsmax*8*sizeof(Limb))
364 GC.@preserve x begin
365 limbs = UnsafeView(x.d, sp.nlimbs)
366 while true
367 rand!(rng, limbs)
368 limbs[end] &= sp.mask
369 MPZ.mpn_cmp(x, sp.m, sp.nlimbs) <= 0 && break
370 end
371 # adjust x.size (normally done by mpz_limbs_finish, in GMP version >= 6)
372 x.size = sp.nlimbs
373 while x.size > 0
374 limbs[x.size] != 0 && break
375 x.size -= 1
376 end
377 end
378 MPZ.add!(x, sp.a)
379 end

106 MersenneTwister(seed)

108
109 Create a `MersenneTwister` RNG object. Different RNG objects can have
110 their own seeds, which may be useful for generating different streams
111 of random numbers.
112 The `seed` may be a non-negative integer or a vector of
113 `UInt32` integers. If no seed is provided, a randomly generated one
114 is created (using entropy from the system).
115 See the [`seed!`](@ref) function for reseeding an already existing
116 `MersenneTwister` object.
117
118
119 # Examples
120 ```jldoctest
121 julia> rng = MersenneTwister(1234);

123 julia> x1 = rand(rng, 2)

126 0.7667970365022592

128 julia> rng = MersenneTwister(1234);

131 2-element Array{Float64,1}:

133 0.7667970365022592

478 # depending on the alignment of A, the data written by fill_array! may have
479 # to be left-shifted by up to 15 bytes (cf. unsafe_copyto! below) for
480 # reproducibility purposes;
481 # so, even for well aligned arrays, fill_array! is used to generate only
482 # the n-2 first values (or n-3 if n is odd), and the remaining values are
483 # generated by the scalar version of rand
484 n = length(A)
485 n2 = (n-2) ÷ 2 * 2
486 n2 < dsfmt_get_min_array_size() && return _rand_max383!(r, A, I[])
487
488 pA = A.ptr
489 align = Csize_t(pA) % 16
490 if align > 0
491 pA2 = pA + 16 - align
492 fill_array!(r.state, pA2, n2, I[]) # generate the data in-place, but shifted
493 unsafe_copyto!(pA, pA2, n2) # move the data to the beginning of the array
494 else
495 fill_array!(r.state, pA, n2, I[])
496 end
497 for i=n2+1:n
498 A[i] = rand(r, I[])
499 end

523 @eval function rand!(r::MersenneTwister, A::Array{$T}, ::SamplerTrivial{CloseOpen12{$T}})
524 n = length(A)
525 n128 = n * sizeof($T) ÷ 16
526 _rand!(r, A, 2*n128, CloseOpen12())
527 GC.@preserve A begin
528 A128 = UnsafeView{UInt128}(pointer(A), n128)
529 for i in 1:n128
530 u = A128[i]
531 u ⊻= u << 26
532 # at this point, the 64 low bits of u, "k" being the k-th bit of A128[i] and "+"
533 # the bit xor, are:
534 # [..., 58+32,..., 53+27, 52+26, ..., 33+7, 32+6, ..., 27+1, 26, ..., 1]
535 # the bits needing to be random are
536 # [1:10, 17:26, 33:42, 49:58] (for Float16)
537 # [1:23, 33:55] (for Float32)
538 # this is obviously satisfied on the 32 low bits side, and on the high side,
539 # the entropy comes from bits 33:52 of A128[i] and then from bits 27:32
540 # (which are discarded on the low side)
541 # this is similar for the 64 high bits of u
542 A128[i] = mask128(u, $T)
543 end
544 end
545 for i in 16*n128÷sizeof($T)+1:n
546 @inbounds A[i] = rand(r, $T) + one($T)
547 end

563 function rand!(r::MersenneTwister, A::UnsafeView{UInt128}, ::SamplerType{UInt128})
564 n::Int=length(A)
565 i = n
566 while true
567 rand!(r, UnsafeView{Float64}(A.ptr, 2i), CloseOpen12())
568 n < 5 && break
569 i = 0
570 while n-i >= 5
571 u = A[i+=1]
572 A[n] ⊻= u << 48
573 A[n-=1] ⊻= u << 36
574 A[n-=1] ⊻= u << 24
575 A[n-=1] ⊻= u << 12
576 n-=1
577 end
578 end
579 if n > 0
580 u = rand(r, UInt2x52Raw())
581 for i = 1:n
582 A[i] ⊻= u << (12*i)
583 end
584 end
585 A
586 end

608 #### arrays of Bool
609
610 # similar to Array{UInt8}, but we need to mask the result so that only the LSB
611 # in each byte can be non-zero
612
613 function rand!(r::MersenneTwister, A1::Array{Bool}, sp::SamplerType{Bool})
614 n1 = length(A1)
615 n128 = n1 ÷ 16
616
617 if n128 == 0
618 bits = rand(r, UInt52Raw())
619 else
620 GC.@preserve A1 begin
621 A = UnsafeView{UInt128}(pointer(A1), n128)
622 rand!(r, UnsafeView{Float64}(A.ptr, 2*n128), CloseOpen12())
623 # without masking, non-zero bits could be observed in other
624 # positions than the LSB of each byte
625 mask = 0x01010101010101010101010101010101
626 # we need up to 15 bits of entropy in `bits` for the final loop,
627 # which we will extract from x = A[1] % UInt64;
628 # let y = x % UInt32; y contains 32 bits of entropy, but 4
629 # of them will be used for A[1] itself (the first of
630 # each byte). To compensate, we xor with (y >> 17),
631 # which gets the entropy from the second bit of each byte
632 # of the upper-half of y, and sets it in the first bit
633 # of each byte of the lower half; the first two bytes
634 # now contain 16 usable random bits
635 x = A[1] % UInt64
636 bits = x ⊻ x >> 17
637 for i = 1:n128
638 # << 5 to randomize the first bit of the 8th & 16th byte
639 # (i.e. we move bit 52 (resp. 52 + 64), which is unused,
640 # to position 57 (resp. 57 + 64))
641 A[i] = (A[i] ⊻ A[i] << 5) & mask
642 end
643 end
644 end
645 for i = 16*n128+1:n1
646 @inbounds A1[i] = bits % Bool
647 bits >>= 1
648 end
649 A1
650 end

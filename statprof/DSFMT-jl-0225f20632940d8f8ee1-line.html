103 "Represents a polynomial in GF(2)[X]"
104 struct GF2X
105 z::BigInt
106 end
107
108 GF2X(s::AbstractString) = GF2X(parse(BigInt, reverse(s), base = 16))
109 Base.string(f::GF2X) = reverse(string(f.z, base = 16))
110 Base.:(==)(f::GF2X, g::GF2X) = f.z == g.z
111 Base.copy(f::GF2X) = GF2X(MPZ.set(f.z))
112
113 degree(f::GF2X) = Base.ndigits0z(f.z, 2) - 1
114 coeff(f::GF2X, i) = MPZ.tstbit(f.z, i)
115 setcoeff!(f::GF2X, i) = (MPZ.setbit!(f.z, i); f)
116
117 mulx!(f::GF2X, c::Int=1) = (MPZ.mul_2exp!(f.z, c), f)
118 xor!(f::GF2X, g::GF2X) = (MPZ.xor!(f.z, g.z); f)

120 # compute f*X mod m
121 function mulxmod!(f::GF2X, m::GF2X, deg=degree(m))::GF2X
122 # precondition: degree(f) < degree(m)
123 mulx!(f)
124 degree(f) == deg && xor!(f, m)
125 f
126 end

128 # cache for X^(2i) mod m
129 const _squares = Dict{GF2X, Vector{GF2X}}()
130
131 # compute f^2 mod m
132 function sqrmod!(f::GF2X, m::GF2X)::GF2X
133 d = degree(m)-1
134 0 <= degree(f) <= d || throw(DomainError("f must satisfy 0 <= degree(f) <= degree(m)-1"))
135 sqrs = get!(_squares, m) do
136 x2i = GF2X(1)
137 GF2X[copy(mulxmod!(mulxmod!(x2i, m, d+1), m, d+1)) for i=1:d]
138 end
139 foldl(filter(i->coeff(f, i), 0:degree(f)); init=GF2X(0)) do g, i
140 i <= d÷2 ? # optimization for "simple" squares
141 setcoeff!(g, 2i) :
142 xor!(g, sqrs[i])
143 end
144 end

146 # compute X^e mod m
147 function powxmod(e::BigInt, m::GF2X)::GF2X
148 e < 0 && throw(DomainError("e must be >= 0"))
149 foldl(Base.ndigits0z(e, 2)-1:-1:0; init=GF2X(1)) do f, i
150 MPZ.tstbit(e, i) ?
151 mulxmod!(sqrmod!(f, m), m) :
152 sqrmod!(f, m)
153 end

189 function dsfmt_jump(s::DSFMT_state, jp::GF2X)
190 degree(jp) < degree(CharPoly()) || throw(DomainError("degree of jump polynomial must be < $(degree(CharPoly())), got $(degree(jp))"))
191
192 val = s.val
193 nval = length(val)
194 index = val[nval - 1]
195 work = zeros(Int32, JN32)
196 rwork = reinterpret(UInt64, work)
197 dsfmt = Vector{UInt64}(undef, nval >> 1)
198 ccall(:memcpy, Ptr{Cvoid}, (Ptr{UInt64}, Ptr{Int32}, Csize_t),
199 dsfmt, val, (nval - 1) * sizeof(Int32))
200 dsfmt[end] = UInt64(N*2)
201
202 for i in 0:degree(jp)
203 coeff(jp, i) && dsfmt_jump_add!(rwork, dsfmt)
204 dsfmt_jump_next_state!(dsfmt)
205 end
206
207 rwork[end] = index
208 return DSFMT_state(work)
209 end

211 function dsfmt_jump_add!(dest::AbstractVector{UInt64}, src::Vector{UInt64})
212 dp = dest[end] >> 1
213 sp = src[end] >> 1
214 diff = ((sp - dp + N) % N)
215 i = 1
216 while i <= N-diff
217 j = i*2-1
218 p = j + diff*2
219 dest[j] ⊻= src[p]
220 dest[j+1] ⊻= src[p+1]
221 i += 1
222 end
223 while i <= N
224 j = i*2-1
225 p = j + (diff - N)*2
226 dest[j] ⊻= src[p]
227 dest[j+1] ⊻= src[p+1]
228 i += 1
229 end
230 dest[N*2+1] ⊻= src[N*2+1]
231 dest[N*2+2] ⊻= src[N*2+2]
232 return dest
233 end

235 function dsfmt_jump_next_state!(mts::Vector{UInt64})
236 POS1 = 117
237 SL1 = 19
238 SR = 12
239 MSK1 = 0x000ffafffffffb3f
240 MSK2 = 0x000ffdfffc90fffd
241
242 idx = (mts[end] >> 1) % N
243
244 a = idx*2+1
245 b = ((idx + POS1) % N)*2+1
246 u = N*2+1
247
248 t0 = mts[a]
249 t1 = mts[a+1]
250 L0 = mts[u]
251 L1 = mts[u+1]
252 mts[u] = xor(t0 << SL1, L1 >> 32, L1 << 32, mts[b])
253 mts[u+1] = xor(t1 << SL1, L0 >> 32, L0 << 32, mts[b+1])
254 mts[a] = xor(mts[u] >> SR, mts[u] & MSK1, t0)
255 mts[a+1] = xor(mts[u+1] >> SR, mts[u+1] & MSK2, t1)
256
257 mts[end] = (mts[end] + 2) % (N*2)
258 return mts
259 end

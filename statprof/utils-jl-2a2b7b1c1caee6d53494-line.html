116 function eigsigns(eigs::Vector{<: Real}, tol::Real)
117 dp = count(x -> tol < x, eigs) # number of positive eigenvalues
118 dz = count(x -> -tol < x < tol, eigs) # number of numerically zero eigenvalues
119 dn = count(x -> x < -tol, eigs) # number of negative eigenvalues
120 return dp, dz, dn
